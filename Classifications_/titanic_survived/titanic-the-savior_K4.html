from sklearn.preprocessing import OneHotEncoder
def OH_Custom(df):
  OH_encoder = OneHotEncoder(handle_unknown='ignore', sparse=False)
  OH_Cat= pd.DataFrame(OH_encoder.fit_transform(df.loc[:,['Sex','Embarked']]),columns=['Sex_M','Sex_F','Embarked_C','Embarked_Q','Embarked_S'])
  df=df.join(OH_Cat)
  df.drop(['Sex','Embarked'],axis=1,inplace=True)
  return df
